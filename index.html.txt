  <script>
    const CSV_URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vSnpX_1E_Rl8WpKjJYA-qECY9gc9XizO51MHSagv-9Lm2ckEmLVypx6b254ONVK5Z4ieAEE74hmcFbZ/pub?gid=0&single=true&output=csv";
    
    // ATENÇÃO: COLOQUE AQUI O LINK QUE O RENDER TE DEU
    const RENDER_API_URL = "https://botsgp.onrender.com/chat"; 
    const WHATSAPP_NUM = "5561992282334";
    
    let faqData = [];
    let conversationHistory = "";
    let userName = "";

    function normalize(t) { return t.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").trim(); }

    function addMsg(txt, type, isHTML = false) {
      const chat = document.getElementById("chat-area");
      const div = document.createElement("div");
      div.className = "msg " + type;
      if (isHTML) div.innerHTML = txt; else div.textContent = txt;
      chat.appendChild(div);
      chat.scrollTop = chat.scrollHeight;
      conversationHistory += (type === 'user' ? "Usuário: " : "SGP: ") + (isHTML ? "[Mídia]" : txt) + "\n";
    }

    async function startApp() {
      userName = document.getElementById("nome").value.trim();
      if (!userName) return alert("Informe seu nome.");
      document.getElementById("view-1").style.display = "none";
      document.getElementById("view-2").style.display = "flex";
      
      try {
        const r = await fetch(CSV_URL);
        const csv = await r.text();
        // Colunas: Pergunta, Chaves, Resposta, Categoria, Subcategoria
        faqData = csv.split("\n").slice(1).map(l => {
          const c = l.split(",");
          return { p: c[1], k: c[2], r: c[3], cat: c[4], sub: c[5] };
        }).filter(f => f.r);
        addMsg(`Olá ${userName}, sou o assistente do SGP-SEDE turbinado com Gemini. Como posso ajudar?`, "bot");
      } catch(e) {
        addMsg("Erro ao carregar base de conhecimento.", "bot");
      }
    }

    async function onSend() {
      const input = document.getElementById("chat-input");
      const val = input.value.trim();
      if (!val) return;

      addMsg(val, "user");
      input.value = "";
      
      // Mostrar indicador de digitação
      const typingIndicator = document.getElementById("typing-indicator");
      typingIndicator.style.display = "block";

      try {
        const response = await fetch(RENDER_API_URL, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            pergunta: val,
            nome: userName,
            base_conhecimento: faqData
          })
        });

        const data = await response.json();
        typingIndicator.style.display = "none";
        
        if (data.resposta) {
          renderFinalResponse(data.resposta);
        } else {
          addMsg("Desculpe, tive um problema para processar sua dúvida.", "bot");
        }
      } catch (error) {
        typingIndicator.style.display = "none";
        addMsg("Opa! Meu servidor está acordando ou está offline. Tente em alguns segundos.", "bot");
        setTimeout(askWhatsAppFallback, 2000);
      }
    }

    function renderFinalResponse(text) {
      const cleanId = text.trim();
      // Verifica se a resposta é apenas um ID do Youtube
      if (cleanId.length === 11 && !cleanId.includes(" ")) {
        const thumb = `https://img.youtube.com/vi/${cleanId}/hqdefault.jpg`;
        const link = `https://www.youtube.com/watch?v=${cleanId}`;
        addMsg(`<div class="yt-card"><img src="${thumb}"><a href="${link}" target="_blank" class="yt-link-btn">▶ ASSISTIR VÍDEO NO YOUTUBE</a></div>`, "bot", true);
      } else {
        addMsg(text, "bot");
      }
      setTimeout(askMoreHelp, 1500);
    }

    function askMoreHelp() {
      const div = document.createElement("div");
      div.className = "msg bot";
      div.innerHTML = `<div>Resolvi sua dúvida ou precisa de algo mais?</div>
        <div class="choice-row">
          <button class="choice-btn fill" onclick="this.parentElement.parentElement.remove(); addMsg('Qual a sua próxima dúvida?', 'bot')">Nova Dúvida</button>
          <button class="choice-btn" onclick="this.parentElement.parentElement.remove(); askWhatsAppFallback()">Falar com Humano</button>
        </div>`;
      document.getElementById("chat-area").appendChild(div);
      document.getElementById("chat-area").scrollTop = document.getElementById("chat-area").scrollHeight;
    }

    function askWhatsAppFallback() {
      const div = document.createElement("div");
      div.className = "msg bot";
      div.innerHTML = `<div>Deseja encerrar este chat e ir para o WhatsApp com um resumo do que conversamos?</div>
        <div class="choice-row">
          <button class="choice-btn fill" onclick="goToWhatsApp()">Sim, abrir WhatsApp</button>
          <button class="choice-btn" onclick="location.reload()">Não, fechar chat</button>
        </div>`;
      document.getElementById("chat-area").appendChild(div);
      document.getElementById("chat-area").scrollTop = document.getElementById("chat-area").scrollHeight;
    }

    function goToWhatsApp() {
      const resumo = encodeURIComponent(`*SGP-BOT RESUMO*\nUsuário: ${userName}\nMatrícula: ${document.getElementById("matricula").value}\n\n*Histórico:*\n${conversationHistory}`);
      window.open(`https://wa.me/${WHATSAPP_NUM}?text=${resumo}`);
      location.reload();
    }

    document.getElementById("chat-send").onclick = onSend;
    document.getElementById("chat-input").onkeypress = (e) => { if(e.key === 'Enter') onSend(); };
  </script>
